- 시간 복잡도 O(N^2)

function bubbleSort(array) {
  let i, j, temp;
  let result = [...array];
  const length = array.length;

  for (i = 0; i < length; i++) {
    for (j = i; j < length; j++) {
      if (min > array[j]) {
        min = array[j];
        index = j;
      }
    }

    temp = array[i];
    array[i] = array[index];
    array[index] = temp;
  }

  console.log(array);

  return result;
}
