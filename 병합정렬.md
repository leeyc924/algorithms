# 병합정렬

## 특징

- 일단 반으로 나누고 나중에 합쳐서 정렬

- 항상 동일한 시간복잡도를 가진다

- 메모리 활용이 비효율적이다

- 시간 복잡도 O(N \* logN)

## JS

```javascript
let result = []; // 정렬 배열은 반드시 전역 변수로 선언
let count = 0;

function merge(array, m, middle, n) {
  let i = m;
  let j = middle + 1;
  let k = m;

  // 작은 순서대로 배열에 삽입
  while (i <= middle && j <= n) {
    if (array[i] <= array[j]) {
      result[k] = array[i];
      i++;
    } else {
      result[k] = array[j];
      j++;
    }
    k++;
  }

  // 남은 데이터도 삽입
  if (i > middle) {
    for (let t = j; t <= n; t++) {
      result[k] = array[t];
      k++;
    }
  } else {
    for (let t = i; t <= middle; t++) {
      result[k] = array[t];
      k++;
    }
  }

  // 정렬된 배열을 삽입
  for (let t = m; t <= n; t++) {
    array[t] = result[t];
  }
}

function mergeSort(array, m, n) {
  const newArray = [...array];
  // 이외의 경우는 크기가 1개인 경우
  if (m < n) {
    const middle = Math.floor((m + n) / 2);
    mergeSort(newArray, m, middle);
    mergeSort(newArray, middle + 1, n);
    merge(newArray, m, middle, n);
  }
}

const array = [7, 6, 5, 8, 3, 5, 9, 1];
mergeSort(array, 0, array.length - 1);
```
